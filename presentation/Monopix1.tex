        \frametitle{TJ-Monopix1}

        \begin{columns}

                    \hline
                    \hline

                    Pixel size & 36 $\times$ 40 \si{\um\squared}\\
                    Depth & \SI{25}{\um}\\
                    Electrode size & \SI{2}{\um}\\
                    BCID & \SI{40}{MHz} \\
                    ToT-bit & 6 \\
                    Power consumption & $\sim$ 120 \si{mW/cm\squared}\\    
                    \hline
                    \end{tabular}
                \end{table}
            \column{0.6\textwidth}
